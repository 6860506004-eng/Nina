<script>
    class Node {
        constructor(value) {
            this.value = value;
            this.left = null;
            this.right = null;
            this.x = 0;
            this.y = 0;
        }
    }

    let root = null;
    const canvas = document.getElementById('tree-canvas');

    function handleAdd() {
        const input = document.getElementById('inputNode');
        const val = parseInt(input.value);
        if (isNaN(val)) return;

        root = insert(root, val);
        input.value = '';
        render();
    }

    function insert(node, val) {
        if (!node) return new Node(val);
        if (val < node.value) node.left = insert(node.left, val);
        else if (val > node.value) node.right = insert(node.right, val);
        return node;
    }

    function handleReset() {
        root = null;
        render();
    }

    function render() {
        canvas.innerHTML = '';
        if (!root) {
            updateTraversals();
            return;
        }
        calculatePositions(root, canvas.offsetWidth / 2, 50, canvas.offsetWidth / 4);
        drawLines(root);
        drawNodes(root);
        updateTraversals();
    }

    function calculatePositions(node, x, y, spacing) {
        if (!node) return;
        node.x = x;
        node.y = y;
        calculatePositions(node.left, x - spacing, y + 70, spacing / 1.8);
        calculatePositions(node.right, x + spacing, y + 70, spacing / 1.8);
    }

    function drawLines(node) {
        if (!node) return;
        if (node.left) {
            createLine(node.x, node.y, node.left.x, node.left.y);
            drawLines(node.left);
        }
        if (node.right) {
            createLine(node.x, node.y, node.right.x, node.right.y);
            drawLines(node.right);
        }
    }

    function createLine(x1, y1, x2, y2) {
        const line = document.createElement('div');
        line.className = 'connector';
        const length = Math.sqrt((x2-x1)**2 + (y2-y1)**2);
        const angle = Math.atan2(y2-y1, x2-x1) * 180 / Math.PI;
        line.style.width = length + 'px';
        line.style.left = x1 + 'px';
        line.style.top = y1 + 'px';
        line.style.transform = `rotate(${angle}deg)`;
        canvas.appendChild(line);
    }

    function drawNodes(node) {
        if (!node) return;
        const div = document.createElement('div');
        div.className = 'node';
        div.style.left = (node.x - 21) + 'px';
        div.style.top = (node.y - 21) + 'px';
        div.innerText = node.value;
        canvas.appendChild(div);
        drawNodes(node.left);
        drawNodes(node.right);
    }

    function updateTraversals() {
        let pre = [], ino = [], post = [];
        const traverse = (n) => {
            if (!n) return;
            pre.push(n.value);
            traverse(n.left);
            ino.push(n.value);
            traverse(n.right);
            post.push(n.value);
        };
        traverse(root);
        document.getElementById('preorder').innerText = pre.join(', ') || '-';
        document.getElementById('inorder').innerText = ino.join(', ') || '-';
        document.getElementById('postorder').innerText = post.join(', ') || '-';
    }

    document.getElementById('inputNode').addEventListener('keypress', e => {
        if (e.key === 'Enter') handleAdd();
    });
</script>
